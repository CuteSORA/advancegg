Get Started in Minutes

1

Install AdvanceGG

go get github.com/GrandpaEJ/advancegg
2

Create Your First Graphic

package main

import "github.com/GrandpaEJ/advancegg"

func main() {
    dc := advancegg.NewContext(800, 600)
    
    // Create gradient background
    gradient := advancegg.NewLinearGradient(0, 0, 0, 600)
    gradient.AddColorStop(0, advancegg.ColorFromHex("#667eea"))
    gradient.AddColorStop(1, advancegg.ColorFromHex("#764ba2"))
    dc.SetFillStyle(gradient)
    dc.DrawRectangle(0, 0, 800, 600)
    dc.Fill()
    
    // Draw a circle with shadow
    dc.SetRGBA(0, 0, 0, 0.3)
    dc.DrawCircle(403, 303, 102) // Shadow
    dc.Fill()
    
    dc.SetRGB(1, 0.3, 0.5)
    dc.DrawCircle(400, 300, 100)
    dc.Fill()
    
    // Add text
    dc.LoadFontFace("fonts/arial.ttf", 48)
    dc.SetRGB(1, 1, 1)
    dc.DrawStringAnchored("Hello AdvanceGG!", 400, 400, 0.5, 0.5)
    
    dc.SavePNG("hello.png")
}
3

Run and Enjoy!

go run main.go

Your beautiful graphic is saved as hello.png
